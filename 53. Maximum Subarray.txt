class Solution {
    public int maxSubArray(int[] nums) {
         int max_cur = nums[0], max_so_far = nums[0];
        for(int i=1;i<nums.length;i++)
        {
        	max_cur = Math.max(nums[i], max_cur+nums[i]);
        	max_so_far = Math.max(max_cur, max_so_far);
        }
        return max_so_far;
    }
}
